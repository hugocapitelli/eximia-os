// lib/status-calculator.ts

interface StatusWeight {
  onTrack: number    // 100%
  atRisk: number     // 75%
  offTrack: number   // 25%
  notStarted: number // 0%
  completed: number  // 100%
}

function calculateInitiativeHealth(initiative: Initiative): number {
  const operationals = initiative.operationals

  // Weighted average of operational progress
  const totalWeight = operationals.reduce((sum, op) => {
    return sum + getStatusWeight(op.status)
  }, 0)

  return totalWeight / operationals.length
}

function calculateDriverHealth(driver: Driver): number {
  const initiatives = driver.initiatives
  const kpis = driver.kpis

  // 60% initiatives, 40% KPIs
  const initiativeScore = initiatives.reduce((sum, init) => {
    return sum + calculateInitiativeHealth(init)
  }, 0) / initiatives.length

  const kpiScore = kpis.reduce((sum, kpi) => {
    return sum + (kpi.current / kpi.target) * 100
  }, 0) / kpis.length

  return (initiativeScore * 0.6) + (kpiScore * 0.4)
}

function calculateDimensionHealth(dimension: BSCDimension): number {
  const drivers = getDimensionDrivers(dimension)

  return drivers.reduce((sum, driver) => {
    return sum + calculateDriverHealth(driver)
  }, 0) / drivers.length
}

function getStatusColor(health: number): 'green' | 'yellow' | 'red' {
  if (health >= 80) return 'green'
  if (health >= 60) return 'yellow'
  return 'red'
}

## 9. Real-time Updates

### 9.1 Supabase Realtime Config

```typescript
// hooks/use-war-room-realtime.ts

import { useEffect } from 'react'
import { supabase } from '@/lib/supabase'

export function useWarRoomRealtime(cycleId: string) {
  useEffect(() => {
    // Subscribe to initiative updates
    const initiativeChannel = supabase
      .channel(`initiatives:${cycleId}`)
      .on(
        'postgres_changes',
        {
          event: '*',
          schema: 'public',
          table: 'initiatives',
          filter: `cycle_id=eq.${cycleId}`
        },
        (payload) => {
          // Invalidate React Query cache
          queryClient.invalidateQueries(['war-room', cycleId])
        }
      )
      .subscribe()

    // Subscribe to KPI value updates
    const kpiChannel = supabase
      .channel(`kpi_values:${cycleId}`)
      .on(
        'postgres_changes',
        {
          event: 'INSERT',
          schema: 'public',
          table: 'kpi_values'
        },
        (payload) => {
          queryClient.invalidateQueries(['war-room', cycleId])
        }
      )
      .subscribe()

    return () => {
      supabase.removeChannel(initiativeChannel)
      supabase.removeChannel(kpiChannel)
    }
  }, [cycleId])
}
```
