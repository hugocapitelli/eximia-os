    // Notify team when package is ready
    this.eventBus.on('handoff:package:ready', async (event) => {

        type: 'handoff_ready',
        title: 'Handoff Package Ready',
        message: `${event.contents.storyCount} stories ready for export`,
        recipients: ['dev_team', 'pm'],
      });
    });

    // Update stories when exported to GitHub
    this.eventBus.on('handoff:github:issues_created', async (event) => {
      this.eventBus.emit('backlog:stories:github_linked', {
        packageId: event.packageId,
        repository: event.repository,
        issueCount: event.issueCount,
      });
    });

    // Track export metrics
    this.eventBus.on('handoff:export:completed', async (event) => {
      this.eventBus.emit('analytics:track', {
        event: 'handoff_export',
        properties: {
          destination: event.destination,
          packageId: event.packageId,
        },
      });
    });

    // Alert on export failure
    this.eventBus.on('handoff:export:failed', async (event) => {
      this.eventBus.emit('notification:send', {
        type: 'error',
        title: 'Export Failed',
        message: `Failed to export to ${event.destination}: ${event.error}`,
        recipients: ['pm'],
      });
    });
  }
}
