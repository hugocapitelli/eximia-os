-- Update epic progress when story status changes
CREATE OR REPLACE FUNCTION update_epic_progress()
RETURNS TRIGGER AS $$
BEGIN
    UPDATE backlog_epics
    SET
        completed_points = (
            SELECT COALESCE(SUM(story_points), 0)
            FROM backlog_stories
            WHERE epic_id = NEW.epic_id AND status = 'done'
        ),
        progress = (
            SELECT
                CASE
                    WHEN SUM(story_points) > 0 THEN
                        (SUM(CASE WHEN status = 'done' THEN story_points ELSE 0 END)::DECIMAL / SUM(story_points)) * 100
                    ELSE 0
                END
            FROM backlog_stories
            WHERE epic_id = NEW.epic_id
        ),
        updated_at = NOW()
    WHERE id = NEW.epic_id;

    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

    AFTER INSERT OR UPDATE OF status, story_points ON backlog_stories
    FOR EACH ROW EXECUTE FUNCTION update_epic_progress();

interface EstimationAssistant {
  name: 'estimation_assistant';

  capabilities: [
    'suggest_points',
    'find_similar_stories',
    'explain_estimate',
    'detect_estimation_drift'
  ];

  algorithm: {
    // Find similar stories using embeddings
    findSimilar: async (story: UserStory): Promise<SimilarStory[]> => {
      // Generate embedding for story
      const embedding = await generateEmbedding(
        `${story.asA} ${story.iWant} ${story.soThat} ${story.description}`
      );

      // Search vector DB for similar completed stories
      const similar = await vectorDB.search({
        embedding,
        filter: {
          status: 'done',
          hasPoints: true,
        },
        limit: 5,
      });

      return similar.map(s => ({
        storyId: s.id,
        title: s.title,
        points: s.storyPoints,
        similarity: s.score,
      }));
    };

    // Suggest estimate based on similar stories
    suggestEstimate: (similarStories: SimilarStory[]): AIEstimate => {
      const weightedSum = similarStories.reduce(
        (sum, s) => sum + s.points * s.similarity,
        0
      );
      const totalWeight = similarStories.reduce(
        (sum, s) => sum + s.similarity,
        0
      );

      const rawEstimate = weightedSum / totalWeight;
      const fibonacci = [1, 2, 3, 5, 8, 13, 21];
      const nearest = fibonacci.reduce((prev, curr) =>
        Math.abs(curr - rawEstimate) < Math.abs(prev - rawEstimate) ? curr : prev
      );

      return {
        value: nearest,
        confidence: calculateConfidence(similarStories),
        reasoning: generateReasoning(similarStories, nearest),
        similarStories,
      };
    };
  };
}
