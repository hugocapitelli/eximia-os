interface FlowAnalyzer {
  name: 'flow_analyzer';

  capabilities: [
    'detect_dead_ends',
    'find_missing_screens',
    'suggest_happy_path',
    'identify_complexity'
  ];

  analyze: async (document: WireframeDocument): Promise<FlowAnalysis> => {
    const screens = document.screens;
    const connections = document.connections;

    return {
      deadEnds: findDeadEndScreens(screens, connections),
      orphanScreens: findOrphanScreens(screens, connections),
      suggestedConnections: suggestMissingConnections(screens),
      complexityScore: calculateFlowComplexity(connections),
      happyPath: identifyMainFlow(screens, connections),
      recommendations: generateRecommendations(),
    };
  };
}

  complexityScore: number;      // 1-10

}
