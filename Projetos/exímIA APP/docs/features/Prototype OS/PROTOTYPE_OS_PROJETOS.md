-- Projetos
CREATE TABLE product_projects (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  organization_id UUID NOT NULL REFERENCES organizations(id),

  name VARCHAR(200) NOT NULL,
  slug VARCHAR(200) NOT NULL,
  description TEXT,
  project_type VARCHAR(30) NOT NULL,

  phase VARCHAR(30) NOT NULL DEFAULT 'ideation',
  priority VARCHAR(20) NOT NULL DEFAULT 'medium',
  progress INTEGER DEFAULT 0,

  start_date DATE NOT NULL DEFAULT CURRENT_DATE,
  target_date DATE,
  completed_at TIMESTAMPTZ,

  owner_id UUID NOT NULL REFERENCES team_members(id),

  tags JSONB DEFAULT '[]',
  parent_project_id UUID REFERENCES product_projects(id),
  linked_initiative_id UUID,

  enabled_phases JSONB NOT NULL,

  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW(),
  created_by UUID NOT NULL REFERENCES team_members(id),

  UNIQUE(organization_id, slug)
);

-- Membros do projeto
CREATE TABLE project_members (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  project_id UUID NOT NULL REFERENCES product_projects(id) ON DELETE CASCADE,
  user_id UUID NOT NULL REFERENCES team_members(id),

  role VARCHAR(20) NOT NULL DEFAULT 'contributor',
  added_at TIMESTAMPTZ DEFAULT NOW(),
  added_by UUID NOT NULL REFERENCES team_members(id),

  UNIQUE(project_id, user_id)
);

-- Artefatos
CREATE TABLE project_artifacts (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  project_id UUID NOT NULL REFERENCES product_projects(id) ON DELETE CASCADE,

  artifact_type VARCHAR(30) NOT NULL,
  name VARCHAR(200) NOT NULL,
  status VARCHAR(20) NOT NULL DEFAULT 'pending',
  version VARCHAR(20) DEFAULT '1.0',

  url TEXT,
  linked_entity_id UUID,
  linked_entity_type VARCHAR(30),

  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- Atividade
CREATE TABLE project_activities (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  project_id UUID NOT NULL REFERENCES product_projects(id) ON DELETE CASCADE,

  activity_type VARCHAR(30) NOT NULL,
  actor_id UUID NOT NULL REFERENCES team_members(id),

  details JSONB,

  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Templates
CREATE TABLE project_templates (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  organization_id UUID REFERENCES organizations(id),

  name VARCHAR(100) NOT NULL,
  description TEXT,
  project_type VARCHAR(30) NOT NULL,

  phases JSONB NOT NULL,
  default_artifacts JSONB DEFAULT '[]',
  phase_checklists JSONB DEFAULT '{}',

  suggested_duration INTEGER, -- dias
  suggested_team_size INTEGER,

  is_default BOOLEAN DEFAULT FALSE,

  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- ComentÃ¡rios
CREATE TABLE project_comments (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  project_id UUID NOT NULL REFERENCES product_projects(id) ON DELETE CASCADE,

  author_id UUID NOT NULL REFERENCES team_members(id),
  content TEXT NOT NULL,
  parent_id UUID REFERENCES project_comments(id),

  artifact_id UUID REFERENCES project_artifacts(id),

  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- Ãndices
CREATE INDEX idx_projects_org_phase ON product_projects(organization_id, phase);
