```typescript
type SkillCategory = 'business' | 'design' | 'tech' | 'marketing' | 'leadership' | 'soft_skills';
type SkillStatus = 'locked' | 'available' | 'in_progress' | 'unlocked';

interface Skill {
  id: string;
  name: string;
  description: string;
  category: SkillCategory;
  icon: string;

  // Hierarchy
  level: number;                    // Current level (0 = not unlocked)
  maxLevel: number;                 // Max achievable level
  parentSkillIds: string[];         // Prerequisites
  childSkillIds: string[];          // Skills this unlocks

  // Requirements
  requiredSkills: SkillRequirement[];
  requiredXP?: number;

  // Progress
  status: SkillStatus;
  progress?: number;                // 0-100 if in_progress
  currentCourseId?: string;         // Curso que estÃ¡ desenvolvendo esta skill

  // Unlock info
  unlockedAt?: Date;
  unlockedVia?: string;             // Course ID that unlocked it

  // Position (for tree visualization)
  treePosition: { x: number; y: number };
}

interface SkillRequirement {
  skillId: string;
  minLevel: number;
}

interface UserSkillProfile {
  totalUnlocked: number;
  totalAvailable: number;
  totalSkills: number;
  inProgress: number;
  strongestCategory: SkillCategory;
  suggestedNextSkill?: Skill;
  xpTotal: number;
}

interface SkillTreePageData {
  profile: UserSkillProfile;
  skills: Skill[];
  categories: SkillCategory[];
  connections: SkillConnection[];   // Para desenhar as linhas do tree
}

interface SkillConnection {
  fromSkillId: string;
  toSkillId: string;
  isUnlocked: boolean;
}
```
