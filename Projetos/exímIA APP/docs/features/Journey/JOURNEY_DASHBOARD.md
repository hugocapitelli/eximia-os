## TypeScript Interfaces

```typescript
interface JourneyDashboard {
  user: User;
  date: Date;

  // Widgets Data
  todayFocus: Goal | null;
  weekScore: WeekScore;
  habitsToday: HabitToday[];
  readingProgress: ReadingProgress;
  goalsOverview: GoalsOverview;
  upcomingEvents: CalendarEvent[];
}

interface WeekScore {
  total: number;           // 0-100
  habitsScore: number;     // 0-100
  goalsScore: number;      // 0-100
  readingScore: number;    // 0-100
  trend: number;           // vs last week (+/-)
  status: 'excellent' | 'good' | 'average' | 'needs_improvement';
}

interface HabitToday {
  id: string;
  name: string;
  completed: boolean;
  streak: number;
  scheduledTime?: string;
}

interface ReadingProgress {
  currentBook: Book | null;
  currentProgress: number;      // 0-100
  currentPage: number;
  totalPages: number;
  lastReadAt: Date;
  yearlyGoal: number;
  yearlyCompleted: number;
}

interface GoalsOverview {
  quarterly: { completed: number; total: number; percentage: number };
  monthly: { completed: number; total: number; percentage: number };
  weekly: { completed: number; total: number; percentage: number };
}

interface CalendarEvent {
  id: string;
  title: string;
  startTime: Date;
  duration: number;        // minutes
  type: 'meeting' | 'focus' | 'personal' | 'deadline';
  isToday: boolean;
}
```
