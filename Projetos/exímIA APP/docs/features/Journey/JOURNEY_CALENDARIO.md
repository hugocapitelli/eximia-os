```typescript
type EventType = 'meeting' | 'focus' | 'habit' | 'deadline' | 'personal' | 'reminder';

interface CalendarEvent {
  id: string;
  title: string;
  description?: string;
  type: EventType;
  source: EventSource;

  // Timing
  startTime: Date;
  endTime: Date;
  allDay: boolean;

  color?: string;

  createdAt: Date;
  updatedAt: Date;
}

  type: 'focus';
  category: 'deep_work' | 'shallow_work' | 'admin' | 'creative';

}

  events: CalendarEvent[];

    connected: boolean;
    lastSync?: Date;
    calendarIds: string[];
  };
}

interface HabitSchedule {
  habitId: string;
  habitName: string;
  time: string;                   // HH:MM

  habitsScheduled: number;

  deadlines: GoalDeadline[];
}
```
