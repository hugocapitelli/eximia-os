```typescript
function calculateENPS(responses: number[]): ENPSResult {
  const promoters = responses.filter(r => r >= 9).length;
  const passives = responses.filter(r => r >= 7 && r <= 8).length;
  const detractors = responses.filter(r => r <= 6).length;

  const total = responses.length;
  const enps = ((promoters - detractors) / total) * 100;

  return {
    score: Math.round(enps),
    promotersPercent: Math.round((promoters / total) * 100),
    passivesPercent: Math.round((passives / total) * 100),
    detractorsPercent: Math.round((detractors / total) * 100),
    totalResponses: total,
    interpretation: interpretENPS(enps)
  };
}

function interpretENPS(score: number): string {
  if (score >= 70) return 'Excelente';
  if (score >= 50) return 'Muito Bom';
  if (score >= 30) return 'Bom';
  if (score >= 0) return 'AtenÃ§Ã£o';
  return 'CrÃ­tico';
}
```
