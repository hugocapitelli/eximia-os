## TypeScript Interfaces

```typescript
type AssetCategory =
  | 'logos'
  | 'icons'
  | 'images'
  | 'patterns'
  | 'templates'
  | 'social_media'
  | 'presentations'
  | 'documents';

type AssetFormat = 'svg' | 'png' | 'jpg' | 'pdf' | 'psd' | 'ai' | 'pptx' | 'docx' | 'html' | 'fig';

interface BrandAsset {
  id: string;
  name: string;
  description?: string;
  category: AssetCategory;
  tags: string[];

  // Primary file
  primaryFile: AssetFile;

  // Variations (different formats/sizes)
  variations: AssetFile[];

  // Guidelines
  usageGuidelines?: string[];

  // Metadata
  createdAt: Date;
  updatedAt: Date;
  createdBy: string;
  updatedBy?: string;

  // Stats
  downloadCount: number;
}

interface AssetFile {
  id: string;
  name: string;
  format: AssetFormat;
  mimeType: string;
  url: string;
  size: number;                   // bytes
  dimensions?: {
    width: number;
    height: number;
  };
  isVector: boolean;
  dpi?: number;
}

interface AssetFolder {
  id: string;
  name: string;
  parentId?: string;
  category: AssetCategory;
  assetCount: number;
  createdAt: Date;
}

interface AssetFilters {
  search?: string;
  category?: AssetCategory[];
  format?: AssetFormat[];
  tags?: string[];
  dateRange?: {
    start: Date;
    end: Date;
  };
}

interface AssetCategoryConfig {
  id: AssetCategory;
  name: string;
  icon: string;
  allowedFormats: AssetFormat[];
  maxFileSize: number;            // bytes
}

interface AssetsPageData {
  assets: BrandAsset[];
  folders: AssetFolder[];
  categories: AssetCategoryConfig[];
  filters: AssetFilters;
  totalAssets: number;
  storageUsed: number;
  storageLimit: number;
}

interface AssetUpload {
  files: File[];
  name: string;
  category: AssetCategory;
  tags: string[];
  description?: string;
  folderId?: string;
}
```
