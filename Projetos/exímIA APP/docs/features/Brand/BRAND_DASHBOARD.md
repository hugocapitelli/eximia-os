## TypeScript Interfaces

```typescript
interface BrandIdentity {
  id: string;

  // Basic Info
  name: string;
  tagline: string;
  description?: string;

  // Mission, Vision, Values
  mission: string;
  vision: string;
  values: BrandValue[];

  // Visual Identity
  logo: LogoAsset;
  colors: ColorPalette;
  typography: Typography;

  // Brand Voice (summary)
  voiceSummary: BrandVoiceSummary;

  // Assets
  quickAssets: QuickAsset[];

  // Applications
  applications: BrandApplication[];

  // Meta
  updatedAt: Date;
  updatedBy: string;
}

interface BrandValue {
  id: string;
  name: string;
  description: string;
  icon?: string;
  order: number;
}

interface LogoAsset {
  id: string;
  name: string;
  primaryUrl: string;
  variations: LogoVariation[];
}

interface LogoVariation {
  id: string;
  name: string;               // "Horizontal", "Vertical", "Icon Only"
  colorMode: 'full_color' | 'monochrome' | 'white' | 'black';
  format: 'svg' | 'png' | 'pdf';
  url: string;
  width?: number;
  height?: number;
}

interface ColorPalette {
  primary: Color[];
  secondary: Color[];
  neutral: Color[];
  semantic: SemanticColors;
  gradients?: Gradient[];
}

interface Color {
  id: string;
  name: string;
  hex: string;
  rgb: { r: number; g: number; b: number };
  hsl?: { h: number; s: number; l: number };
  usage: string;
  cssVariable?: string;
}

interface SemanticColors {
  success: string;
  warning: string;
  danger: string;
  info: string;
}

interface Gradient {
  id: string;
  name: string;
  colors: string[];
  direction: string;
  css: string;
}

interface Typography {
  primary: FontFamily;
  secondary?: FontFamily;
  display?: FontFamily;
  mono?: FontFamily;
  scale: TypeScale;
}

interface FontFamily {
  name: string;
  category: 'sans-serif' | 'serif' | 'monospace' | 'display';
  weights: number[];
  googleFontsUrl?: string;
  localFiles?: string[];
  usage: string;
  sample: string;
}

interface TypeScale {
  xs: { size: string; lineHeight: string };
  sm: { size: string; lineHeight: string };
  base: { size: string; lineHeight: string };
  lg: { size: string; lineHeight: string };
  xl: { size: string; lineHeight: string };
  '2xl': { size: string; lineHeight: string };
  '3xl': { size: string; lineHeight: string };
  '4xl': { size: string; lineHeight: string };
}

interface BrandVoiceSummary {
  toneAttributes: string[];
  personality: PersonalityTrait[];
}

interface PersonalityTrait {
  trait: string;
  icon: string;
  description: string;
}

interface QuickAsset {
  id: string;
  name: string;
  type: 'logo' | 'icon' | 'favicon' | 'pattern' | 'illustration';
  previewUrl: string;
  downloadUrl: string;
  format: string;
}

interface BrandApplication {
  id: string;
  name: string;
  icon: string;
  description: string;
  thumbnailUrl?: string;
  guidelineUrl?: string;
}

interface BrandDashboardData {
  identity: BrandIdentity;
  recentUpdates: BrandUpdate[];
}

interface BrandUpdate {
  id: string;
  section: string;
  description: string;
  updatedAt: Date;
  updatedBy: string;
}
```
