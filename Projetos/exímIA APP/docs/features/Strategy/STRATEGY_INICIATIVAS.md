```typescript
interface InitiativeDetail extends Initiative {
  // Full description
  fullDescription?: string;

  // Milestones
  milestones: Milestone[];

  // Tasks (pode linkar com Journey Goals)
  tasks: InitiativeTask[];

  // Dependencies
  blockedBy: string[];              // IDs de outras iniciativas
  blocks: string[];                 // IDs que esta bloqueia

  // Resources
  estimatedEffort?: number;         // Em horas ou story points
  actualEffort?: number;
  budget?: number;

  // Updates
  updates: InitiativeUpdate[];

  // Cascade
  linkedGoals: LinkedGoal[];        // Goals no Journey
}

interface Milestone {
  id: string;
  title: string;
  dueDate: Date;
  completed: boolean;
  completedAt?: Date;
}

interface InitiativeTask {
  id: string;
  title: string;
  completed: boolean;
  assignee?: string;
  linkedGoalId?: string;            // Se cascadeou para Journey
}

interface InitiativeUpdate {
  id: string;
  content: string;
  progress?: number;                 // Se atualizou progresso
  createdBy: string;
  createdAt: Date;
}

interface LinkedGoal {
  goalId: string;
  goalTitle: string;
  progress: number;
  syncEnabled: boolean;              // Se sincroniza progresso automaticamente
}
```
