interface RoadmapItem {
  id: string;
  type: 'initiative' | 'milestone' | 'deadline';
  title: string;
  description?: string;

  // Timing
  startDate: Date;
  endDate: Date;
  duration: number;                  // days

  // Status
  progress: number;
  status: 'planned' | 'in_progress' | 'completed' | 'at_risk' | 'behind';
  priority: 'critical' | 'high' | 'medium' | 'low';

  // Visual
  color?: string;
  icon?: string;

  // Dependencies
  dependsOn?: string[];              // IDs de outras iniciativas
  blockedBy?: string[];

  // Links
  initiativeId?: string;
  objectiveId?: string;

  // Milestones
  milestones?: RoadmapMilestone[];
}

  id: string;
  title: string;
  date: Date;
  completed: boolean;
}

  startDate: Date;
  endDate: Date;
  zoom: 'day' | 'week' | 'month' | 'quarter';
  items: RoadmapItem[];
  cycleMilestones: RoadmapMilestone[];
  todayLine: Date;
}

interface RoadmapPageData {
  currentCycleId: string;
  currentCycleName: string;
  view: RoadmapView;
  viewMode: 'gantt' | 'timeline' | 'kanban';
  filters: RoadmapFilters;
}

interface RoadmapFilters {
  status?: string[];
  priority?: string[];
  owner?: string[];
  objective?: string[];
