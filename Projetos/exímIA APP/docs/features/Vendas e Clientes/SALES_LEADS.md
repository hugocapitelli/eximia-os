```sql
BEGIN
    -- Update counters based on activity type

    VALUES (NEW.lead_id)
    ON CONFLICT (lead_id) DO UPDATE SET
        email_opens = CASE WHEN NEW.type = 'email_opened'

            ELSE lead_engagement.email_opens END,
        email_clicks = CASE WHEN NEW.type = 'email_clicked'

    RETURN NEW;
END;
```

```typescript
interface LeadScoringAI {
  name: 'lead_scoring_ai';

  calculateScore: async (leadId: string): Promise<LeadScore> => {
    const lead = await getLead(leadId);
    const config = await getScoringConfig(lead.organizationId);
    const engagement = await getLeadEngagement(leadId);

    let fitScore = 0;
    let behaviorScore = 0;
    const breakdown: ScoreBreakdown = { fit: [], behavior: [] };

    // Apply fit rules
    for (const rule of config.fitRules) {
      if (evaluateRule(rule, lead)) {
        fitScore += rule.points;
        breakdown.fit.push({
          rule: rule.field,
          value: getFieldValue(lead, rule.field),
          points: rule.points,
          reason: rule.description,
        });
      }
    }

    // Apply behavior rules
    for (const rule of config.behaviorRules) {
      if (evaluateRule(rule, engagement)) {
        behaviorScore += rule.points;
        breakdown.behavior.push({
          rule: rule.field,
          value: getFieldValue(engagement, rule.field),
          points: rule.points,
          reason: rule.description,
        });
      }
    }

    // Apply negative rules
    for (const rule of config.negativeRules) {
      if (evaluateRule(rule, { ...lead, ...engagement })) {
        behaviorScore += rule.points; // negative points
        breakdown.behavior.push({
          rule: rule.field,
          value: getFieldValue({ ...lead, ...engagement }, rule.field),
          points: rule.points,
          reason: rule.description,
        });
      }
    }

    const totalScore = Math.max(0, Math.min(100, fitScore + behaviorScore));
    const temperature = getTemperature(totalScore, config.thresholds);

    return {
      total: totalScore,
      fitScore: Math.min(fitScore, config.maxFitScore),
      behaviorScore: Math.min(Math.max(behaviorScore, 0), config.maxBehaviorScore),
      temperature,
      breakdown,
      lastCalculatedAt: new Date(),
    };
  };
}
```

```typescript
interface LeadEnrichmentAI {

    // Enrich from email domain
    const companyData = await enrichFromDomain(lead.email.split('@')[1]);

    let linkedinData = null;
    if (lead.linkedinUrl) {
      linkedinData = await enrichFromLinkedIn(lead.linkedinUrl);
    }

    const enrichedData: Partial<Lead> = {
      companyName: companyData?.name || lead.companyName,
      companyIndustry: companyData?.industry || lead.companyIndustry,
      companySize: companyData?.size || lead.companySize,
      companyDomain: companyData?.domain || lead.companyDomain,
      jobTitle: linkedinData?.title || lead.jobTitle,
      department: linkedinData?.department || lead.department,
    };

    return {
      leadId,
      enrichedFields: Object.keys(enrichedData).filter(
        k => enrichedData[k] !== lead[k]
      ),
      data: enrichedData,
      sources: [
        companyData ? 'domain_lookup' : null,
        linkedinData ? 'linkedin' : null,
      ].filter(Boolean),
```
