    // Process into segments
    const segments = processSegments(rawTranscript);

    // Identify speakers
    const identifiedSegments = await identifySpeakers(segments, callId);

    return {
      callId,
      fullText: identifiedSegments.map(s => s.text).join(' '),
      segments: identifiedSegments,
      wordCount: rawTranscript.words.length,
      speakerCount: new Set(identifiedSegments.map(s => s.speaker)).size,
      language: 'pt-BR',
      confidence: rawTranscript.confidence,
      processedAt: new Date(),
    };
  };
