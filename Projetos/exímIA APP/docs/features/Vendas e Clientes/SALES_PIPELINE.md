    this.eventBus.on('pipeline:deal:dragged', async (event) => {

interface DealScoringEngine {
  name: 'deal_scoring';

  calculateWinProbability: async (dealId: string): Promise<WinProbability> => {
    const deal = await getDeal(dealId);
    const company = await getCompany(deal.companyId);
    const activities = await getAllActivities(dealId);
    const historicalData = await getHistoricalWinRates();

    // Factor 1: Stage probability (base)
    const stageProbability = getStageBaseProbability(deal.stage);

    // Factor 2: Activity engagement
    const engagementScore = calculateEngagementScore(activities);

    // Factor 3: Company fit
    const fitScore = calculateFitScore(company);

    // Factor 4: Velocity (how fast is the deal moving?)
    const velocityScore = calculateVelocityScore(deal);

    // Factor 5: Historical patterns
    const patternScore = matchHistoricalPatterns(deal, historicalData);

    // Weighted combination
    const probability = (
      stageProbability * 0.3 +
      engagementScore * 0.25 +
      fitScore * 0.2 +
      velocityScore * 0.15 +
      patternScore * 0.1
    );

    return {
      probability: Math.round(probability),
      factors: {
        stage: stageProbability,
        engagement: engagementScore,
        fit: fitScore,
        velocity: velocityScore,
        pattern: patternScore,
      },
      insights: generateProbabilityInsights(deal, { engagementScore, fitScore, velocityScore }),
    };
  };
}

  factors: {
