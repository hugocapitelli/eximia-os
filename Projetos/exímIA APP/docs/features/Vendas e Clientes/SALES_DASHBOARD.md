    // Celebrate deal wins
    this.eventBus.on('sales:deal:won', async (event) => {
      this.eventBus.emit('notification:send', {
        type: 'celebration',
        title: 'ğŸ‰ Deal Fechado!',
        message: `${event.ownerName} fechou ${event.name} por R$ ${event.value.toLocaleString()}`,
        recipients: ['sales_team'],
      });

      // Update finance module
      this.eventBus.emit('finance:revenue:recorded', {
        dealId: event.dealId,
        amount: event.value,
        type: 'deal_closed',
      });
    });

    // Alert on at-risk deals
    this.eventBus.on('sales:deal:at_risk', async (event) => {
      if (event.riskScore > 70) {
        this.eventBus.emit('notification:send', {
          type: 'warning',
          title: 'âš ï¸ Deal em Risco',
          message: `${event.name} (R$ ${event.value.toLocaleString()}) precisa de atenÃ§Ã£o`,
          recipients: ['sales_manager', event.ownerId],
        });
      }
    });

    // Track email engagement
    this.eventBus.on('sales:email:opened', async (event) => {
      if (event.openCount >= 3) {
        this.eventBus.emit('sales:ai:insight_generated', {
          insightId: `insight_${Date.now()}`,
          type: 'opportunity',
          priority: 'high',
          relatedEntityId: event.dealId,
        });
      }
    });

    // Update deal on lost reason for analytics
    this.eventBus.on('sales:deal:lost', async (event) => {
      this.eventBus.emit('analytics:track', {
        event: 'deal_lost',
        properties: {
          dealId: event.dealId,
          value: event.value,
          reason: event.reason,
          competitor: event.competitor,
        },
      });
    });
  }
}

CREATE OR REPLACE FUNCTION track_deal_stage_change()
RETURNS TRIGGER AS $$
BEGIN
    IF OLD.stage IS DISTINCT FROM NEW.stage THEN
        INSERT INTO sales_deal_stage_history (
            deal_id, from_stage, to_stage, changed_by, time_in_previous_stage
        )
        VALUES (
            NEW.id,
            OLD.stage,
            NEW.stage,
            auth.uid(),
            EXTRACT(EPOCH FROM (NOW() - COALESCE(
                (SELECT changed_at FROM sales_deal_stage_history
                 WHERE deal_id = NEW.id ORDER BY changed_at DESC LIMIT 1),
                OLD.created_at
            ))) / 3600
        );
    END IF;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

CREATE TRIGGER trigger_track_deal_stage_change
    AFTER UPDATE ON sales_deals
    FOR EACH ROW EXECUTE FUNCTION track_deal_stage_change();

interface RevenueForecaster {
  name: 'revenue_forecaster';

  generateForecast: async (organizationId: string, period: MetricPeriod): Promise<RevenueForecast> => {
    const pipeline = await getPipelineDeals(organizationId);
    const historicalData = await getHistoricalRevenue(organizationId);
    const winRates = await calculateWinRates(organizationId);

    // Calculate committed (high probability deals)
    const committed = pipeline
      .filter(d => d.probability >= 80)
      .reduce((sum, d) => sum + d.value * (d.probability / 100), 0);

    // Calculate best case
    const bestCase = pipeline
      .filter(d => d.probability >= 50 && d.probability < 80)
      .reduce((sum, d) => sum + d.value * (d.probability / 100), 0);

    // Calculate upside
    const upside = pipeline
      .filter(d => d.probability >= 20 && d.probability < 50)
      .reduce((sum, d) => sum + d.value * (d.probability / 100), 0);

    // AI confidence based on historical accuracy
    const aiConfidence = calculateConfidence(historicalData, winRates);

    return {
      type: 'realistic',
      period,
      summary: { committed, bestCase, upside, total: committed + bestCase + upside },
      aiConfidence,
      confidenceFactors: analyzeConfidenceFactors(pipeline, historicalData),
    };
