## 4. Database Schema

```sql
-- =============================================================================
-- PROPOSAL MANAGEMENT - DATABASE SCHEMA
-- =============================================================================

-- -----------------------------------------------------------------------------
-- Proposals
-- -----------------------------------------------------------------------------

CREATE TABLE sales_proposals (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    organization_id UUID NOT NULL,

    -- Basic info
    name VARCHAR(300) NOT NULL,
    proposal_number VARCHAR(50) NOT NULL,
    status VARCHAR(30) NOT NULL DEFAULT 'draft',

    -- Related
    deal_id UUID REFERENCES sales_deals(id),
    contact_id UUID REFERENCES sales_contacts(id),
    company_id UUID REFERENCES sales_companies(id),

    -- Owner
    created_by UUID NOT NULL REFERENCES auth.users(id),

    -- Content
    sections JSONB NOT NULL DEFAULT '[]',
    template_id UUID REFERENCES proposal_templates(id),

    -- Pricing
    pricing JSONB NOT NULL DEFAULT '{}',

    -- Validity
    valid_until DATE NOT NULL,
    sent_at TIMESTAMPTZ,
    viewed_at TIMESTAMPTZ,
    responded_at TIMESTAMPTZ,

    -- Response
    accepted_at TIMESTAMPTZ,
    declined_at TIMESTAMPTZ,
    decline_reason TEXT,

    -- Signature
    signature_required BOOLEAN DEFAULT FALSE,
    signed_at TIMESTAMPTZ,
    signed_by VARCHAR(200),
    signature_url TEXT,

    -- Sharing
    share_url TEXT,
    share_password VARCHAR(100),

    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),

    UNIQUE (organization_id, proposal_number)
);

-- Templates
-- -----------------------------------------------------------------------------

CREATE TABLE proposal_templates (

    default_pricing JSONB,
```
