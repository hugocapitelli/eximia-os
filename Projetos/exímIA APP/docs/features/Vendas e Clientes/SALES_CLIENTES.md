interface CustomerHealthScorer {
  name: 'customer_health_scorer';

  calculateHealth: async (customerId: string): Promise<HealthScore> => {
    const customer = await getCustomer(customerId);
    const config = await getHealthConfig(customer.organizationId);
    const metrics = await collectHealthMetrics(customerId);

    const breakdown: HealthBreakdown = {
      productUsage: calculateProductUsage(metrics),
      engagement: calculateEngagement(metrics),
      support: calculateSupport(metrics),
      payment: calculatePayment(metrics),
      growth: calculateGrowth(metrics),
    };

    const totalScore = Object.entries(breakdown).reduce((sum, [key, metric]) => {
      return sum + (metric.score * config.weights[key] / 100);
    }, 0);

    const status = getHealthStatus(totalScore, config.thresholds);

    return {
      score: Math.round(totalScore),
      status,
      breakdown,
      calculatedAt: new Date(),
    };
  };
}

interface ExpansionDetector {
  name: 'expansion_detector';

  detectOpportunities: async (customerId: string): Promise<ExpansionOpportunity[]> => {
    const customer = await getCustomer(customerId);
    const usage = await getUsageMetrics(customerId);
    const activities = await getRecentActivities(customerId);
    const conversations = await getRecentConversations(customerId);

    const opportunities: ExpansionOpportunity[] = [];

    // Seat expansion
    if (usage.activeUsers / customer.totalSeats > 0.85) {
      opportunities.push({
        type: 'expansion',
        title: 'Seat Expansion',
        description: `Cliente utilizando ${Math.round(usage.activeUsers / customer.totalSeats * 100)}% dos seats`,
        estimatedValue: calculateSeatExpansionValue(customer),
        probability: 80,
        source: 'usage_trigger',
      });
    }

    // Feature upsell based on usage patterns
    const underutilizedFeatures = findUnderutilizedFeatures(usage);
    const potentialFeatures = matchFeaturesToNeeds(underutilizedFeatures, customer);
    for (const feature of potentialFeatures) {
      opportunities.push({
        type: 'upsell',
        title: `Upsell: ${feature.name}`,
        description: feature.reasoning,
        estimatedValue: feature.value,
        probability: feature.probability,
        source: 'ai_detected',
      });
    }

    // Cross-sell based on conversation analysis
    const mentionedNeeds = await analyzeConversationsForNeeds(conversations);
    for (const need of mentionedNeeds) {
      opportunities.push({
        type: 'cross_sell',
        title: need.title,
        description: need.context,
        estimatedValue: need.estimatedValue,
        probability: need.probability,
        source: 'ai_detected',
        triggerData: { conversationId: need.conversationId },
      });
    }

    return opportunities;
  };
}
