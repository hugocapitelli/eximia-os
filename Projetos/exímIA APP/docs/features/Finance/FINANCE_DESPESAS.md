## TypeScript Interfaces

```typescript
type ExpenseType = 'fixed' | 'variable' | 'one_time';
type ExpenseStatus = 'paid' | 'pending' | 'overdue' | 'scheduled';
type PaymentMethod = 'credit_card' | 'debit_card' | 'bank_transfer' | 'pix' | 'boleto' | 'cash';

interface Expense {
  id: string;
  type: ExpenseType;
  description: string;

  // Financial
  amount: number;
  currency: string;
  categoryId: string;

  // Vendor
  vendorName?: string;
  vendorId?: string;

  // Dates
  dueDate: Date;
  paidDate?: Date;
  createdAt: Date;

  // Status
  status: ExpenseStatus;
  paymentMethod?: PaymentMethod;

  // Recurrence (if recurring)
  isRecurring: boolean;
  recurrence?: {
    frequency: RecurrenceFrequency;
    dayOfMonth: number;
    startDate: Date;
    endDate?: Date;
    nextDueDate: Date;
  };

  // Budget tracking
  budgetId?: string;

  // Attachments
  receiptUrl?: string;
  invoiceUrl?: string;

  // Meta
  notes?: string;
  tags?: string[];
}

interface ExpenseCategory {
  id: string;
  name: string;
  icon: string;
  color: string;

  // Budget
  budget: number;
  budgetUsed: number;
  budgetPercent: number;

  // Aggregates
  totalExpenses: number;
  expenseCount: number;
}

interface ExpenseSummary {
  totalMonth: number;
  totalChange: number;
  fixed: number;
  fixedPercent: number;
  variable: number;
  variablePercent: number;
  budgetTotal: number;
  budgetUsed: number;
  budgetPercent: number;
}

interface ExpenseFilters {
  search?: string;
  type?: ExpenseType[];
  status?: ExpenseStatus[];
  categoryId?: string[];
  paymentMethod?: PaymentMethod[];
  dateRange?: {
    start: Date;
    end: Date;
  };
}

interface ExpenseListData {
  expenses: Expense[];
  categories: ExpenseCategory[];
  summary: ExpenseSummary;
}

interface Budget {
  id: string;
  categoryId: string;
  amount: number;
  period: 'monthly' | 'quarterly' | 'yearly';
  alertThreshold: number;        // % quando alertar (ex: 80%)
}
```
