## TypeScript Interfaces

```typescript
type RevenueType = 'recurring' | 'one_time';
type RevenueStatus = 'paid' | 'pending' | 'overdue' | 'cancelled';
type RecurrenceFrequency = 'weekly' | 'monthly' | 'quarterly' | 'yearly';

interface Revenue {
  id: string;
  type: RevenueType;
  description: string;

  // Financial
  amount: number;
  currency: string;
  category: string;

  // Client
  clientId?: string;
  clientName?: string;

  // Dates
  dueDate: Date;
  paidDate?: Date;
  createdAt: Date;

  // Status
  status: RevenueStatus;

  // Recurrence (if recurring)
  isRecurring: boolean;
  recurrence?: {
    frequency: RecurrenceFrequency;
    startDate: Date;
    endDate?: Date;
    nextDueDate: Date;
    occurrences: number;
  };

  // Links
  dealId?: string;
  projectId?: string;
  invoiceId?: string;

  // Meta
  notes?: string;
  tags?: string[];
}

interface RevenueCategory {
  id: string;
  name: string;
  icon: string;
  color: string;
  isRecurring: boolean;
}

interface RevenueSummary {
  totalMonth: number;
  totalChange: number;
  recurring: number;
  recurringPercent: number;
  oneTime: number;
  oneTimePercent: number;
  pending: number;
}

interface RevenueFilters {
  search?: string;
  type?: RevenueType;
  status?: RevenueStatus[];
  category?: string[];
  clientId?: string;
  dateRange?: {
    start: Date;
    end: Date;
  };
}

interface RevenueListData {
  revenues: Revenue[];
  summary: RevenueSummary;
  categories: RevenueCategory[];
  pagination: {
    page: number;
    pageSize: number;
    total: number;
    totalPages: number;
  };
}
```
