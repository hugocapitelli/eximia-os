const calculateProjections = (
  assumptions: ScenarioAssumptions,
  initialValues: { mrr: number; balance: number; expenses: number },
  months: number,
  plannedExpenses: PlannedExpense[]
): MonthlyProjection[] => {
  const projections: MonthlyProjection[] = [];
  let currentMRR = initialValues.mrr;
  let currentBalance = initialValues.balance;
  let currentExpenses = initialValues.expenses;

  for (let i = 0; i < months; i++) {
    const month = new Date();
    month.setMonth(month.getMonth() + i + 1);

    // Calculate revenue
    const churnedRevenue = currentMRR * (assumptions.churnRate / 100);
    const newRevenue = currentMRR * (assumptions.mrrGrowthRate / 100);
    const newMRR = currentMRR + newRevenue - churnedRevenue;

    // Calculate expenses
    const baseExpenses = currentExpenses * (1 + assumptions.expenseGrowthRate / 100);
    const additionalExpenses = getPlannedExpensesForMonth(plannedExpenses, month);
    const totalExpenses = baseExpenses + additionalExpenses;

    // Calculate cash flow
    const netCashFlow = newMRR - totalExpenses;
    const newBalance = currentBalance + netCashFlow;

    projections.push({
      month,
      monthLabel: formatMonth(month),
      mrr: newMRR,
      newRevenue,
      churnedRevenue,
      totalRevenue: newMRR,
      baseExpenses,
      newExpenses: additionalExpenses,
      totalExpenses,
      netCashFlow,
      balance: newBalance,
      runwayMonths: newBalance / totalExpenses,
      burnRate: totalExpenses - newMRR,
    });

    // Update for next iteration
    currentMRR = newMRR;
    currentBalance = newBalance;
    currentExpenses = baseExpenses;
  }

  return projections;
};
