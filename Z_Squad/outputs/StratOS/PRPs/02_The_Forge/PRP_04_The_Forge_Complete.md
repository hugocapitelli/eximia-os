export function useForgeAI() {
  const [isLoading, setIsLoading] = useState(false)
  const [suggestions, setSuggestions] = useState<any[]>([])

  const suggestSWOTItems = async (context: SWOTContext) => {
    setIsLoading(true)
    const result = await aiService.generate({
      mode: 'swot_researcher',
      context
    })
    setSuggestions(result.suggestions)
    setIsLoading(false)
  }

  const evaluateNorthStar = async (type: 'vision' | 'purpose', text: string) => {
    // Debounced evaluation
    const result = await aiService.generate({
      mode: 'northstar_refiner',
      context: { type, text }
    })
    return result.evaluation
  }

  const suggestDrivers = async (context: DriverContext) => {
    setIsLoading(true)
    const result = await aiService.generate({
      mode: 'driver_suggester',
      context
    })
    setSuggestions(result.suggestions)
    setIsLoading(false)
  }

  // ... more methods

  return {
    isLoading,
    suggestions,
    suggestSWOTItems,
    evaluateNorthStar,
    suggestDrivers
  }
}
